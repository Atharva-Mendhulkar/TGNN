def generate_dynamic_graph_type(num_nodes=50, num_edges=150, steps=3, graph_type="ER"):
    if graph_type == "ER":
        G = nx.gnm_random_graph(num_nodes, num_edges)
    elif graph_type == "BA":
        G = nx.barabasi_albert_graph(num_nodes, max(1, num_edges // num_nodes))
    elif graph_type == "WS":
        G = nx.watts_strogatz_graph(num_nodes, k=4, p=0.1)
    
    graphs = [copy.deepcopy(G)]
    for _ in range(steps):
        if random.random() < 0.5 and G.number_of_edges() > 0:
            edge = random.choice(list(G.edges()))
            G.remove_edge(*edge)
        else:
            n1, n2 = random.sample(range(num_nodes), 2)
            G.add_edge(n1, n2)
        graphs.append(copy.deepcopy(G))
    return graphs

# 3. Temporal GCN with GRU memory
class TemporalGCN(nn.Module):
    def __init__(self, in_channels, hidden_channels, num_colors):
        super().__init__()
        self.gcn1 = GCNConv(in_channels, hidden_channels)
        self.gcn2 = GCNConv(hidden_channels, hidden_channels)
        self.gru = nn.GRU(hidden_channels, hidden_channels, batch_first=True)
        self.fc = nn.Linear(hidden_channels, num_colors)
        self.hidden_state = None  # GRU memory

    def forward(self, x, edge_index):
        # x: (num_nodes, feature_dim)
        x = F.relu(self.gcn1(x, edge_index))
        x = F.relu(self.gcn2(x, edge_index))
        x = x.unsqueeze(1)  # (num_nodes, seq_len=1, hidden)
        
        # Reset hidden_state if first step or batch size changed
        if self.hidden_state is None or self.hidden_state.size(1) != x.size(0):
            device = x.device
            self.hidden_state = torch.zeros(1, x.size(0), x.size(2), device=device)
        
        out, self.hidden_state = self.gru(x, self.hidden_state)
        out = out.squeeze(1)  # (num_nodes, hidden)
        logits = self.fc(out)
        return logits

    def reset_hidden(self):
        self.hidden_state = None

def potts_loss(pred_logits, edge_index, recolor_prev=None, lambda_recolor=0.5):
    probs = F.softmax(pred_logits, dim=1)
    
    # Handle empty graph
    if edge_index.size(1) == 0:
        conflict_loss = torch.tensor(0.0, device=pred_logits.device)
    else:
        src, dst = edge_index
        probs_src = probs[src]
        probs_dst = probs[dst]
        conflicts = (probs_src * probs_dst).sum(dim=1)
        conflict_loss = conflicts.mean()
    
    total_loss = conflict_loss
    
    pred = pred_logits.argmax(dim=1)
    if recolor_prev is not None:
        recolor_penalty = lambda_recolor * (pred != recolor_prev).float().mean()
        total_loss = total_loss + recolor_penalty
    
    return total_loss, pred

import time

def hybrid_repair(colors, edge_index, num_colors):
    """
    Quick local repair for conflicting nodes.
    - colors: tensor of predicted colors (num_nodes,)
    - edge_index: PyG edge_index (2, num_edges)
    """
    colors = colors.clone()
    src, dst = edge_index
    conflicts = (colors[src] == colors[dst])
    
    for i, j in zip(src[conflicts], dst[conflicts]):
        # Pick node with fewer same-color neighbors to recolor
        node = i if (colors[i] == colors[j]) else j
        # Try all colors, pick one minimizing conflicts
        min_conflicts = float('inf')
        best_color = colors[node].item()
        for c in range(num_colors):
            colors[node] = c
            new_conflicts = ((colors[src] == colors[dst]).sum().item())
            if new_conflicts < min_conflicts:
                min_conflicts = new_conflicts
                best_color = c
        colors[node] = best_color
    return colors

def count_conflicts(colors, edge_index):
    src, dst = edge_index
    conflicts = (colors[src] == colors[dst]).sum().item()
    return conflicts

def count_recolors(prev_colors, new_colors):
    if prev_colors is None:
        return 0
    return (prev_colors != new_colors).sum().item()

def colors_used(colors):
    return len(colors.unique())
